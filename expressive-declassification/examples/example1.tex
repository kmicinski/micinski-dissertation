(* Policy declarations *)
let L = \x,y -> x,y

let buttonClick(textArea) |>
  let net(reply) |>
    set textArea := reply
  in
  let | loc((l1,l2), L) |> net(l1,l2)

let onCreate() |>
  let newTextArea(a) -o
    let newButton(sendLocation) -o
      (* Wait for the button to be pressed *)
      | let sendLocation() |> buttonClick(a)
    in
    createNewButton()
  in
  createNewTextArea()
